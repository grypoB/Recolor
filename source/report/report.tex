 {
  \For(\tcc*[f]{counts how many times $\frac{n!}{k!}$ can be divided by 2}){$temp1 \leftarrow k+1$ \KwTo $n$}{
   $temp2 \leftarrow temp1$\;
   \While{($temp2 \equiv 0 \pmod 2$) and ($temp2 \neq 0$)}{
    $temp2 \leftarrow \frac{temp2}{2}$\;
    $count \leftarrow count + 1$ \tcc*[r]{store this number in \emph{count}}}
   }
  \For(\tcc*[f]{repeat the process for $(n-k)!$}){$temp1 \leftarrow 2$ \KwTo $n-k$}{
   $temp2 \leftarrow temp1$\;
   \While{($temp2 \equiv 0 \pmod 2$) and ($temp2 \neq 0$)}{
    $temp2 \leftarrow \frac{temp2}{2}$\;
    $count \leftarrow count - 1$ \tcc*[r]{decrement \emph{count} by this ammount}}
   }
 }
 \BlankLine
 \eIf{$count > 0$}{\KwRet{true}\;}
 {\KwRet{false}\;}
}
